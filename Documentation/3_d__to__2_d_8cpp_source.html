102 void get_planes(){
104  float x1,x2,y1,y2,z1,z2,p,q,r;
105  for(int i=0; i<number_of_faces; i++){ //take first three coordinates XCor[0][i]
106  x1=Xcor[0][i]-Xcor[1][i];
107  x2=Xcor[1][i]-Xcor[2][i];

110  z1=Zcor[0][i]-Zcor[1][i];
111  z2=Zcor[1][i]-Zcor[2][i];
112  p=y1*z2-y2*z1;
113  q=-1.0f*(x1*z2-x2*z1);
114  r=x1*y2-x2*y1;
115  planes[0][i] = p;
116  planes[1][i] = q;
117  planes[2][i] = r;
118  planes[3][i] = -1*(p*Xcor[0][i] + q*Ycor[0][i] + r*Zcor[0][i]);
